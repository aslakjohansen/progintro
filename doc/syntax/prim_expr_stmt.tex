  \SyntaxWestSplit{MainWest}
  \SyntaxEastSplit{MainEast}

    \node[nonterminal] (ruleI) {name};
    &
    &
    \node[terminal]    (ruleIa) {++};
    &
    &
    \\
    &
    &
    \node[terminal]    (ruleIb) {--};
    &
    &
    \\
    &
    &
    \node[terminal]    (ruleIIa) {++};
    &
    &
    \node[nonterminal] (ruleII) {name};
    \\
    &
    &
    \node[terminal]    (ruleIIb) {--};
    &
    &
    \\
  };
  
  \draw[path] (begin)--(ruleI)--(ruleIa)--(end);
  \draw[path] (begin)--(ruleI) to[-|-] (ruleIb) to[-|-] (end);
  \draw[path] (begin)--([xshift=-2cm]ruleI.west) to[-|-] ([xshift=-3cm]ruleIIa.west)--(ruleIIa) -- (ruleII) to[-|-] (end);
  \draw[path] (begin)--([xshift=-2cm]ruleI.west) to[-|-] ([xshift=-3cm]ruleIIb.west)--(ruleIIb) to[-|-] (ruleII) to[-|-] (end);
\end{syntax}
