\begin{syntax}{class-annotations}
  \SyntaxWestSplit{MainWest}
  \SyntaxEastSplit{MainEast}
  
  \draw[path] (begin)--(end);
\end{syntax}

\begin{syntax}{parameter-list}
  \SyntaxWestSplit{MainWest}
  \SyntaxEastSplit{MainEast}
  
  \node[sequence] () at ([yshift=-1*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[nonterminal] (ruleIa) {type};
    &
    \node[nonterminal] (ruleIb) {name};
    \\
  };
  
  \node[sequence] () at ([yshift=-2*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[terminal]    (ruleII) {,};
    \\
  };
  
  \draw[path] (begin)--(end);
  
  \draw[path]
    (begin)
    -| ([xshift=-2cm,yshift=0.4cm]ruleIa.west) |-
    (ruleIa)--(ruleIb)
    -| ([xshift=2cm,yshift=0.4cm]ruleIb.east) |-
    (end);
  
  \draw[path] (ruleIb)
  -| ([xshift=1cm,yshift=-0.4cm]ruleIb.east) |- (ruleII) -| ([xshift=-1cm,yshift=-0.4cm]ruleIa.west)
  |- (ruleIa);
\end{syntax}

\begin{syntax}{class-body}
  \SyntaxWestSplit{MainWest}
  \SyntaxEastSplit{MainEast}
  
  \node[sequence] () at ([yshift=-1*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[nonterminal]    (ruleI) {context-variable};
    \\
  };
  
  \node[sequence] () at ([yshift=-2*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[nonterminal]    (ruleII) {constructor};
    \\
  };
  
  \node[sequence] () at ([yshift=-3*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[nonterminal]    (ruleIII) {method};
    \\
  };
  
  \draw[path] (begin)--(end);
  
  \draw[path]
    (begin)
    -| ([xshift=3cm,yshift=-0.4cm]begin.east) |-
    (ruleI)
    -| ([xshift=-3cm,yshift=-0.4cm]end.west) |-
    (end);
  \draw[path]
    (begin)
    -| ([xshift=3cm,yshift=-0.4cm]begin.east) |-
    (ruleII)
    -| ([xshift=-3cm,yshift=-0.4cm]end.west) |-
    (end);
  \draw[path]
    (begin)
    -| ([xshift=3cm,yshift=-0.4cm]begin.east) |-
    (ruleIII)
    -| ([xshift=-3cm,yshift=-0.4cm]end.west) |-
    (end);
  
  \draw[path] (begin) -| ([xshift=-2cm,yshift=0.4cm]end.west) |- ([yshift=1*\syntaxruledist]$(begin)!0.5!(end)$) -| ([xshift=2cm,yshift=0.4cm]begin.east) |- (end);
\end{syntax}

\begin{syntax}{class}
  \SyntaxWestSplit{MainWest}
  \SyntaxEastSplit{MainEast}
  
  \node[sequence] () at ([yshift=-0*\syntaxruledist]$(begin)!0.4!(end)$) {
    \node[nonterminal] (ruleIa) {class-annotations};
    &
    \node[terminal]    (ruleIb) {class};
    &
    \node[nonterminal] (ruleIc) {name};
    \\
  };
  
  \node[sequence] () at ([yshift=-1.5*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[terminal]    (ruleId) {(};
    &
    \node[nonterminal] (ruleIe) {parameter-list};
    &
    \node[terminal]    (ruleIf) {)};
    \\
  };
  
  \node[sequence] () at ([yshift=-3*\syntaxruledist]$(begin)!0.5!(end)$) {
    \node[terminal] (ruleIi) {\{};
    &
    \node[nonterminal] (ruleIj) {class-body};
    &
    \node[terminal] (ruleIk) {\}};
    \\
  };
  
  \draw[path] (begin)--(ruleIa)--(ruleIb)--(ruleIc)
    -|([xshift=1cm,yshift=-0.4cm]ruleIc.east) to[|-|] ([xshift=-1cm,yshift=0.4cm]ruleId.west)|-
    (ruleId)--(ruleIe)--(ruleIf)
    -|([xshift=1cm,yshift=-0.4cm]ruleIf.east) to[|-|] ([xshift=-1cm,yshift=0.4cm]ruleIi.west)|-
    (ruleIi)--(ruleIj)--(ruleIk) to[-|-] (end);
\end{syntax}
