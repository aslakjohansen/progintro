               -- (seqE)
               -- (seqF)

      \node[endpoint] (condEnd) at ([yshift=-7.5*\spacing]condAnchor) {};

      \path[flow,-] (condBegin)

      \path[flow,-,draw=purple] (condB)

      \path[flow] (condE)
               -- (condF)
               -- (condEnd)
      ;
    }

               to [out=270,in=90] (forkE)
      ;
      \path[flow] (forkE)
               -- (forkF)
               -- (forkEnd)
      ;
    }

    % while loop
    {
      \node[endpoint] (whileBegin) at ([yshift=0*\spacing]whileAnchor) {};
      \node[stmt] (whileA) at ([yshift=-1*\spacing]whileAnchor) {};
      \node[stmt,choice] (whileB) at ([yshift=-2*\spacing]whileAnchor) {};
      \node[stmt,draw=purple] (whileC) at ([yshift=-2.5*\spacing,xshift=0.5*\spacing]whileAnchor) {};
      \node[stmt,draw=purple] (whileD) at ([yshift=-2.5*\spacing,xshift=1.5*\spacing]whileAnchor) {};
      \node[stmt] (whileE) at ([yshift=-3*\spacing]whileAnchor) {};
      \node[stmt] (whileF) at ([yshift=-4*\spacing]whileAnchor) {};
      \node[endpoint] (whileEnd) at ([yshift=-5.5*\spacing]whileAnchor) {};
      \node[label] (while) at ([yshift=0*\spacing]whileEnd) {Skippable Loop};
      
      \path[flow] (whileBegin)
               -- (whileA)
               -- (whileB)
               -- (whileE)
               -- (whileF)

      \path[flow,-,draw=purple](whileC)

      ;
